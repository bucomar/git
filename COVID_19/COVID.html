def prep(url):
    # Index=Countries, Columns=Datum, Values=Values
    df = pd.read_csv(url)
    df.set_index('Country/Region', inplace=True)
    df.drop(columns=['Province/State', 'Lat', 'Long'], inplace=True)
    return df
